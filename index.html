豆包AI辅助 - 拼音作业检测代码示例

# 拼音作业检测核心代码（Python）
# 集成豆包API实现拼音发音和书写检测
import requests

# 豆包API配置（示例）
DOUBAO_API_KEY = "your_api_key"
DOUBAO_API_URL = "https://api.doubao.com/pinyin/check"

def check_pinyin_homework(homework_content):
    """
    检测幼儿拼音作业：包含发音准确性、书写规范性、声调正确性
    :param homework_content: 作业内容（如拼音书写文本/音频转写）
    :return: 检测结果和AI辅导建议
    """
    headers = {
        "Authorization": f"Bearer {DOUBAO_API_KEY}",
        "Content-Type": "application/json"
    }
    data = {
        "content": homework_content,
        "type": "children_pinyin",  # 幼儿拼音专项检测
        "age_group": "3-6"  # 适配3-6岁幼儿认知水平
    }
    
    response = requests.post(DOUBAO_API_URL, json=data, headers=headers)
    result = response.json()
    
    # 解析检测结果
    if result["code"] == 200:
        check_result = {
            "pronunciation_error": result["data"]["pronunciation_error"],  # 发音错误
            "tone_error": result["data"]["tone_error"],  # 声调错误
            "writing_error": result["data"]["writing_error"],  # 书写错误
            "ai_suggestion": result["data"]["ai_suggestion"]  # AI辅导建议
        }
        return check_result
    else:
        return {"error": "检测失败，请重试"}

# 示例调用
if __name__ == "__main__":
    # 幼儿作业示例：书写的拼音文本
    child_homework = "bō lè ǘ"
    result = check_pinyin_homework(child_homework)
    print("拼音作业检测结果：")
    print(f"发音错误：{result.get('pronunciation_error', '无')}")
    print(f"声调错误：{result.get('tone_error', '无')}")
    print(f"书写错误：{result.get('writing_error', '无')}")
    print(f"AI辅导建议：{result.get('ai_suggestion', '无')}")
